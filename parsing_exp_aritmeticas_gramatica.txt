Exp ::= ITE

ITE ::= if ITE then DIS else DIS
| DIS

DIS ::= CON Disjunction

Disjunction ::= 'or' CON Disjunction
| empty

CON ::= B Conjunction

Conjunction ::= 'and' B Conjunction
| empty

B ::= P Bool

Bool ::= '==' P Bool
| '<=' P Bool
| '<' P Bool
| empty

P ::= M Plus

Plus ::= '+' M Plus
| '-' M Plus
| empty

M ::= F Mul

Mul ::= '*' F Mul
| '/' F Mul
| empty

F ::= num | True | False | var
| '~' F
| '(' Exp ')'
| 'let' name '<-' Exp 'in' Exp 'end'
| 'not' F
